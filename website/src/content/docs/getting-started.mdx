<!-- JS amélioré pour fermer le menu mobile -->
<script>
document.addEventListener('DOMContentLoaded', function() {
    function initNavbar() {
        const navbarCollapse = document.querySelector('#navbarCollapse');
        const navLinks = document.querySelectorAll('.navbar-nav .nav-link');
        const navbarToggler = document.querySelector('.navbar-toggler');
        
        if (!navbarCollapse || !navLinks.length || !navbarToggler) {
            setTimeout(initNavbar, 100);
            return;
        }
        
        navLinks.forEach(function(link) {
            link.addEventListener('click', function(e) {
                const isMobile = window.getComputedStyle(navbarToggler).display !== 'none';
                
                if (isMobile && navbarCollapse.classList.contains('show')) {
                    setTimeout(function() {
                        try {
                            if (typeof bootstrap !== 'undefined' && bootstrap.Collapse) {
                                const bsCollapse = new bootstrap.Collapse(navbarCollapse, {
                                    toggle: false
                                });
                                bsCollapse.hide();
                            } else {
                                navbarCollapse.classList.remove('show');
                            }
                        } catch (error) {
                            navbarCollapse.classList.remove('show');
                        }
                    }, 150);
                }
            });
        });
    }
    
    initNavbar();
});
</script>
